import pandas as pd
import numpy as np

# Example DataFrame
data = {
    'Company': ['Company SARL', 'Another Company', 'One More EURL', 'SPA Company', 'XYZ Inc']
}
df = pd.DataFrame(data)

# Define your keywords
keywords = ['SARL', 'EURL', 'SPA']

# Create a new column based on keyword presence
df['Type'] = np.where(df['Company'].str.contains('|'.join(keywords), case=False, na=False), 'PM', 'INC')

# Display the updated DataFrame
print(df)
